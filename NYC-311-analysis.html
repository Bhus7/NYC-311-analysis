from scipy.stats import f_oneway

top5_complaints = df['Complaint Type'].value_counts().head(5).index

# Convert timedelta to hours inside the loop
samples = [
    df[df['Complaint Type'] == complaint]['Request_Closing_Time'].dt.total_seconds() / 3600
    for complaint in top5_complaints
]

# Perform ANOVA
f_stat, p_val = f_oneway(*samples)

print("F-statistic:", f_stat)
print("P-value:", p_val)
